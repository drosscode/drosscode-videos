    Public Function GetNextJobNumber() As Object
        Dim db As Database = DatabaseFactory.CreateDatabase()
        Using cmd As DbCommand = db.GetStoredProcCommand(&quot;CORPHIER_PKG.get_job_no&quot;)
            db.AddParameter(cmd, &quot;retval&quot;, DbType.Int32, 0, ParameterDirection.ReturnValue, True, 0, 0, String.Empty, DataRowVersion.Current, Convert.DBNull)
            db.ExecuteNonQuery(cmd)
            Return db.GetParameterValue(cmd, &quot;retval&quot;)
        End Using
    End Function
